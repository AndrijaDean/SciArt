/** * DAISY Radar Diagram * ─────────────────── * Self-contained module. Call init(containerSelector, values?) to render. * * Usage: * import { init, update } from './daisy-diagram.js'; * init('#my-container'); // render with default values (all 50) * update({ eq1n: 80, eq1f: 90, ... }); // push new values and redraw * * Values object keys: * eq1n/f, eq2n/f, eq3n/f, eq4n/f — Equity spokes (now / 2050) * bi1n/f, bi2n/f, bi3n/f, bi4n/f — Biodiversity spokes * tr1n/f, tr2n/f, tr3n/f — Transformation spokes * All values 0–100. Default 50 = neutral. */ const SPOKES = [ { n: 'eq1', lbl: 'Procedural\nequity', c: '#D85A30' }, { n: 'eq2', lbl: 'Distributive\nequity', c: '#D85A30' }, { n: 'eq3', lbl: 'Recognitional\nequity', c: '#D85A30' }, { n: 'eq4', lbl: 'Capabilities\nequity', c: '#D85A30' }, { n: 'bi1', lbl: 'Genetic\ndiversity', c: '#C0392B' }, { n: 'bi2', lbl: 'Species\ndiversity', c: '#C0392B' }, { n: 'bi3', lbl: 'Habitat\ndiversity', c: '#C0392B' }, { n: 'bi4', lbl: 'Ecosystem\nfunctions', c: '#C0392B' }, { n: 'tr1', lbl: 'Personal\nlevel', c: '#534AB7' }, { n: 'tr2', lbl: 'Practical\nlevel', c: '#534AB7' }, { n: 'tr3', lbl: 'Political\nlevel', c: '#534AB7' }, ]; const GROUPS = [ { ico: '⚖️', lbl: 'Equity', ks: ['eq1', 'eq2', 'eq3', 'eq4'] }, { ico: '🌿', lbl: 'Biodiversity', ks: ['bi1', 'bi2', 'bi3', 'bi4'] }, { ico: '🔄', lbl: 'Transformation', ks: ['tr1', 'tr2', 'tr3'] }, ]; const DEFAULT_VALUES = { eq1n:50, eq1f:50, eq2n:50, eq2f:50, eq3n:50, eq3f:50, eq4n:50, eq4f:50, bi1n:50, bi1f:50, bi2n:50, bi2f:50, bi3n:50, bi3f:50, bi4n:50, bi4f:50, tr1n:50, tr1f:50, tr2n:50, tr2f:50, tr3n:50, tr3f:50, }; const N = SPOKES.length; const CX = 130, CY = 130, R = 88; const SVG_NS = 'http://www.w3.org/2000/svg'; let V = { ...DEFAULT_VALUES }; let svgEl, pnowEl, pfutEl, dlistEl; // ── Geometry ────────────────────────────────────────────────────────────────── function angle(i) { return (Math.PI * 2 * i / N) - Math.PI / 2; } function pt(i, pct) { const a = angle(i); return { x: CX + pct * R * Math.cos(a), y: CY + pct * R * Math.sin(a) }; } function polyPoints(pcts) { return pcts.map((p, i) => { const q = pt(i, p); return `${q.x},${q.y}`; }).join(' '); } // ── Build static skeleton (axes, grid rings, labels) ───────────────────────── function buildSkeleton(svg) { const axesG = svg.querySelector('#raxes'); const labelsG = svg.querySelector('#rlbls'); SPOKES.forEach((s, i) => { const tip = pt(i, 1); // Axis line const line = document.createElementNS(SVG_NS, 'line'); line.setAttribute('x1', CX); line.setAttribute('y1', CY); line.setAttribute('x2', tip.x); line.setAttribute('y2', tip.y); axesG.appendChild(line); // Label const lp = pt(i, 1.28); const lines = s.lbl.split('\n'); const txt = document.createElementNS(SVG_NS, 'text'); txt.setAttribute('x', lp.x); txt.setAttribute('y', lp.y - (lines.length - 1) * 5); txt.setAttribute('fill', s.c); txt.setAttribute('font-weight', '500'); const ta = lp.x < CX - 8 ? 'end' : lp.x> CX + 8 ? 'start' : 'middle'; txt.setAttribute('text-anchor', ta); lines.forEach((ln, li) => { const ts = document.createElementNS(SVG_NS, 'tspan'); ts.setAttribute('x', lp.x); ts.setAttribute('dy', li === 0 ? 0 : '1.15em'); ts.textContent = ln; txt.appendChild(ts); }); labelsG.appendChild(txt); }); // Grid rings at 25%, 50%, 75% [['rg25', 0.25], ['rg50', 0.5], ['rg75', 0.75]].forEach(([id, p]) => { svg.querySelector(`#${id}`).setAttribute('points', polyPoints(SPOKES.map(() => p))); }); } // ── Redraw data polygons ────────────────────────────────────────────────────── function redrawPolygons() { pnowEl.setAttribute('points', polyPoints(SPOKES.map(s => V[s.n + 'n'] / 100))); pfutEl.setAttribute('points', polyPoints(SPOKES.map(s => V[s.n + 'f'] / 100))); } // ── Update group averages table ─────────────────────────────────────────────── function gavg(ks, suf) { return Math.round(ks.reduce((sum, k) => sum + V[k + suf], 0) / ks.length); } function redrawTable() { dlistEl.innerHTML = `
Theme Now 2050
`; GROUPS.forEach(g => { const row = document.createElement('div'); row.className = 'daisy-drow'; row.innerHTML = ` ${g.ico} ${g.lbl} ${gavg(g.ks, 'n')} ${gavg(g.ks, 'f')}`; dlistEl.appendChild(row); }); } // ── DOM template ───────────────────────────────────────────────────────────── function buildHTML(container) { container.innerHTML = `
Daisy diagram
Current
By 2050
`; } // ── Public API ──────────────────────────────────────────────────────────────── /** * Render the diagram inside `selector`. * @param {string} selector CSS selector for the host element * @param {object} [values] Optional initial values (0–100). Missing keys default to 50. */ export function init(selector, values = {}) { const container = document.querySelector(selector); if (!container) throw new Error(`daisy-diagram: no element found for "${selector}"`); buildHTML(container); svgEl = container.querySelector('#daisy-rsvg'); pnowEl = container.querySelector('#pnow'); pfutEl = container.querySelector('#pfut'); dlistEl = container.querySelector('#daisy-dlist'); V = { ...DEFAULT_VALUES, ...values }; buildSkeleton(svgEl); redrawPolygons(); redrawTable(); } /** * Push new values and redraw. Partial updates are fine — only supplied keys change. * @param {object} newValues */ export function update(newValues) { Object.assign(V, newValues); redrawPolygons(); redrawTable(); } /** * Read the current values snapshot. * @returns {object} */ export function getValues() { return { ...V }; }
